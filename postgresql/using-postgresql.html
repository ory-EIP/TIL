# Using PostgreSQL

# PostgreSQL

# postgresql images pull

사용할 PostgreSQL 버전을 선택했다면, 버전에 맞게 postgresql 이미지를 pull받으면 된다.

docker pull postgres:latest

# postgresql docker run

port는 5432(default)로 지정하고 이름과 데이터베이스 비밀번호를 입력하고 타임존은 설정하고 싶다면 -e TZ=Asia/Seoul로 지정하면된다.

docker run -p 5432:5432 --name postgres -e POSTGRES_PASSWORD=12341234 -d postgres:latest

# postgresql setting

# enter the container postgresql

docker exec -it [CONTAINER ID] bash

로그인

psql -U postgres

# database 생성

create database test_db;

# 계정 생성(User) 및 권한 설정

create role test_user with login password 'test1234';

alter user test_user with createdb;
alter user test_user with superuser;

\du

                                   List of roles
 Role name |                         Attributes                         | Member of
-----------+------------------------------------------------------------+-----------
 postgres  | Superuser, Create role, Create DB, Replication, Bypass RLS | {}
 test_user | Superuser, Create DB

# 데이터베이스에 접근 권한 주기

grant all privileges on database test_db to test_user;
